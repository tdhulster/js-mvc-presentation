Don't break your bindings

The wrong way

			
	<div ng-app="">

		<input type="text" ng-model="message">
		<h1>{{message}}</h1>

		<div ng-controller="FirstCtrl">
			<input type="text" ng-model="message">
			<h1>{{message}}</h1>
		</div> 

		<div ng-controller="SecondCtrl">
			<input type="text" ng-model="message">
			<h1>{{message}}</h1>
		</div>

	</div>

The good way

			
	<div ng-app="">

		<input type="text" ng-model="data.message">
		<h1>{{data.message}}</h1>

		<div ng-controller="FirstCtrl">
			<input type="text" ng-model="data.message">
			<h1>{{data.message}}</h1>
		</div> 

		<div ng-controller="SecondCtrl">
			<input type="text" ng-model="data.message">
			<h1>{{data.message}}</h1>
		</div>
		
	</div>
