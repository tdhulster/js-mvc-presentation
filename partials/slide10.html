Use filters

Standard- or custom filters are both reusable and readable

		
			<div ng-repeat="item in myArray|orderBy:name|reverse|unique"></div>
		
	
		
			app.filter('unique', function ()
			{
				return function (items)
				{
					return items.reverse().filter(function (e, i, arr)
					{
					    return arr.indexOf(e, i + 1) === -1;
					}).reverse();
				};
			});
